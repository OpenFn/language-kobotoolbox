Get the list of forms

getForms(state => {
  console.log(state.data);
  return state;
});

Get submissions for a specific form

A query can be used to filter results.

getSubmissions(
  'aXecHjmbATuF6iGFmvBLBX',
  { query: { end: { $gte: '2020-11-20' } } },
  state => {
    console.log(state.data);
    return state;
  }
);
